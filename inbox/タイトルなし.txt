### BUG-01: α成分スコアのNaN → 100満点化

**症状**
- `macro_sensitivity.alpha_20d = NaN` のとき、環境スコア計算式
  `max(0, min(100, 50 + nan * 5.0))` が **100.0** を返している
- NumPyの `nan` は比較でFalseになるため `min(100, nan)` が `nan` となり、
  後段のclipで `100` に丸められている可能性が高い
- 結果、α成分weight=12%が丸ごと加点され、環境スコアが+12ポイント水増し
- 実際の環境スコアは `25.6 - 12 + 0 ≒ 13.6` (実質 🔴 超低品質)

**修正方針**
```python
def compute_alpha_score(alpha_20d: float) -> float:
    """αスコアを環境スコア用に0-100へマップ。NaN時はNeutral=50を返す。"""
    if alpha_20d is None or (isinstance(alpha_20d, float) and math.isnan(alpha_20d)):
        return 50.0  # Neutral (N/A表示をレポート側で明示)
    return max(0.0, min(100.0, 50.0 + alpha_20d * 5.0))
```

### BUG-02: ギャップ予測が欠損ドライバー時に +5.00% 固定化

def predict_gap(drivers: dict, betas: dict, clip_pct: float = 5.0):
    """
    欠損ドライバー対応ギャップ予測。
    - 最大寄与ドライバー(日経)が欠損 → 予測不能として None 返却
    - 2番目以降の欠損 → 該当ドライバーのインパクトを0として継続計算
    """
    # Tier 1: 必須ドライバー(日経)
    if drivers.get('N225_change') is None or math.isnan(drivers['N225_change']):
        return {
            'gap_pct': None,
            'status': 'UNAVAILABLE',
            'reason': 'Primary driver (N225) missing. Cannot forecast.',
            'fallback_suggestion': 'Use CME futures overnight change as proxy (see code TODO).',
        }
    
    # Tier 2: 補助ドライバー(NaN時は0扱い)
    total_impact = 0.0
    breakdown = {}
    for name, beta in betas.items():
        change = drivers.get(f'{name}_change')
        if change is None or math.isnan(change):
            breakdown[name] = {'beta': beta, 'change': 'N/A', 'impact': 0.0}
            continue
        impact = beta * change
        total_impact += impact
        breakdown[name] = {'beta': beta, 'change': change, 'impact': impact}
    
    # 保守的補正 × 0.7
    raw = total_impact
    adjusted = raw * 0.7
    clipped = max(-clip_pct, min(clip_pct, adjusted))
    
    return {
        'gap_pct': clipped,
        'raw': raw,
        'adjusted': adjusted,
        'status': 'OK',
        'breakdown': breakdown,
    }
```

def generate_g0_scenario(gap_info, entry, pmh, gap_fill_prob):
    """G0シナリオ生成。窓埋め確率に応じてTPとRRRを動的に調整。"""
    
    # 窓埋め確率が閾値未満なら TP を段階的に下げる
    if gap_fill_prob < 0.10:  # 10%未満
        # TP = 25%埋め まで (ATR内に収める)
        tp = entry + (pmh - entry) * 0.25
        tp_rationale = f"窓25%埋め (完全埋め確率{gap_fill_prob:.0%}のため段階利確)"
        status = 'DEGRADED'
    elif gap_fill_prob < 0.30:  # 30%未満
        tp = entry + (pmh - entry) * 0.50
        tp_rationale = f"窓50%埋め (完全埋め確率{gap_fill_prob:.0%}のため)"
        status = 'PARTIAL'
    else:
        tp = pmh  # 寄り値回帰を許容
        tp_rationale = "寄り値回帰(完全埋め)"
        status = 'FULL'
    
    sl = entry - atr * 0.20
    rrr = (tp - entry) / (entry - sl)
    
    # RRR < 1.0 なら自動的に NO_TRADE 判定
    if rrr < 1.0:

    return {
        'entry': entry, 'sl': sl, 'tp': tp, 'rrr': rrr,

def enforce_data_freshness(pdc_data: dict, max_staleness_hours: int = 20):
    """データ鮮度FAIL時は生成を停止し、明示的に再取得を試みる。"""
    staleness_hours = (datetime.now(JST) - pdc_data['timestamp']).total_seconds() / 3600
    
    if staleness_hours > max_staleness_hours:
        # 1回だけ再取得を試みる
        fresh_data = refetch_from_multiple_sources(pdc_data['ticker'])
        if fresh_data is None:
            raise DataStalenessError(
                f"PDC data is {staleness_hours:.1f}h old. "
                f"Refresh failed. Cannot generate reliable report."
            )
        return fresh_data
    return pdc_data
```

from enum import Enum
from dataclasses import dataclass
from datetime import datetime

class DataSource(Enum):
    JQUANTS = "J-Quants公式"          # 信頼度 AAA
    YFINANCE_REALTIME = "yfinance(実データ)"   # 信頼度 BBB

    PROXY_CME = "CME先物プロキシ"       # 信頼度 BBB

@dataclass
class DataProvenance:

    fetched_at: datetime
    as_of: datetime  # データが示す時点

    @property
    def is_trustworthy_for_live_trading(self) -> bool:

    def warning_level(self) -> str:

        if self.source == DataSource.YFINANCE_CACHED:

        if self.source == DataSource.UNKNOWN:

        return ""

### AUDIT-02: ギャップ予測の入力ソース・整合性監査 (不整合解消)

**症状**
- CLI手入力で**先物 -1.28%**を渡した形跡
- 一方、本文の CME夜間は **+0.27% (yfinance由来)**
- 最終ギャップは **-0.35%**
- **どの入力をどう合成したかが本文に書かれていない** → 監査不能

**修正要件**
```python
@dataclass
class GapPredictionBreakdown:
    """ギャップ予測の完全な監査証跡"""
    
    # 入力ソース
    inputs: dict  # {driver_name: {value, source, beta, impact, proxy_used}}
    
    # CLI手入力があれば、yfinance値との差分を明示
    cli_overrides: dict | None = None  # {'N225_futures': -1.28} など
    
    # 合成ロジック
    fusion_rule: str = ""  # "CLI手入力を優先 / yfinance不採用"
    
    # 最終出力
    raw_prediction: float = 0.0
    conservative_factor: float = 0.7
    final_prediction: float = 0.0
    
    def render(self) -> str:
        lines = ["### 🔬 ギャップ予測 監査証跡\n"]
        lines.append("| ドライバー | yfinance | CLI入力 | 採用値 | 出所 | β | インパクト |")
        lines.append("|---|---|---|---|---|---|---|")
        
        for name, info in self.inputs.items():
            yf = info.get('yfinance_value', '—')
            cli = info.get('cli_value', '—')
            used = info.get('used_value', '—')
            src = info.get('source', '—')
            beta = info.get('beta', 0)
            impact = info.get('impact', 0)
            
            # 矛盾検出
            mismatch = ""
            if yf != '—' and cli != '—' and abs(float(yf) - float(cli)) > 0.5:
                mismatch = " ⚠️矛盾"
            
            lines.append(f"| {name} | {yf}% | {cli}% | **{used}%**{mismatch} | {src} | {beta:.3f} | {impact:+.2f}% |")
        
        lines.append(f"\n**合成ルール**: {self.fusion_rule}")
        lines.append(f"**生予測**: {self.raw_prediction:+.2f}% × 保守補正{self.conservative_factor} = **{self.final_prediction:+.2f}%**")
        
        return "\n".join(lines)
```

**使用例**
```python

    inputs={
        'N225': {

            'used_value': -1.28,

        },
        'CME_overnight_dummy': {
            'yfinance_value': +0.27,  # 怪しい値
            'cli_value': None,
            'used_value': None,       # 不採用
            'source': '不採用(矛盾検出)',
            'beta': 0,
            'impact': 0,
        },
        # ...
    },
    cli_overrides={'N225_futures': -1.28},
    fusion_rule="CLI手入力(-1.28%)を優先採用。yfinanceの+0.27%と矛盾するため後者不採用。",

```

def render_executive_summary(ctx) -> str:
