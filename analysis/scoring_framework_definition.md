## 3. 判定ロジックの疑似コード

```python
def score_stock(fundamentals, technical, macro, growth, risk):
    """
    銘柄スコアリング判定ロジック
    各カテゴリは独立して採点し、加重和で総合スコアを算出
    """
    
    # STEP 1: カテゴリ別採点
    f_score = (
        earnings_trend_score(fundamentals) * 0.40 +    # 業績トレスト 10/25
        valuation_score(fundamentals) * 0.32 +          # バリュエーション 8/25
        profitability_score(fundamentals) * 0.28         # 収益性 7/25
    )
    
    t_score = (
        trend_score(technical) * 0.40 +                 # トレンド判定 8/20
        momentum_score(technical) * 0.35 +              # モメンタム 7/20
        support_resist_score(technical) * 0.25           # 支持/抵抗線 5/20
    )
    
    m_score = (
        sector_score(macro) * 0.40 +                    # セクター環境 8/20
        fx_rate_score(macro) * 0.35 +                   # 為替・金利 7/20
        geocommodity_score(macro) * 0.25                 # 地政学 5/20
    )
    
    g_score = (
        growth_driver_score(growth) * 0.50 +            # 成長ドライバー 10/20
        innovation_score(growth) * 0.50                  # イノベーション 10/20
    )
    
    r_score = (
        financial_safety_score(risk) * 0.53 +           # 財務安全性 8/15
        resilience_score(risk) * 0.47                    # リスク耐性 7/15
    )
    
    # STEP 2: 総合スコア
    total = f_score + t_score + m_score + g_score + r_score
    
    # STEP 3: 判定
    if total >= 90: return "STRONG_BUY"
    elif total >= 80: return "BUY"
    elif total >= 70: return "NEUTRAL_TO_BUY"
    elif total >= 60: return "NEUTRAL"
    elif total >= 50: return "NEUTRAL_TO_SELL"
    elif total >= 40: return "SELL"
    else: return "STRONG_SELL"
```
